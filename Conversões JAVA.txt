  public static void exemploConversoes() {
        System.out.println("\n--- Conversões de Tipos ---");

        // String → int
        String strIdade = "25";
        int idade = Integer.parseInt(strIdade);

        // int → String
        String idadeStr = String.valueOf(idade);

        // String → double
        String strSalario = "3500.50";
        double salario = Double.parseDouble(strSalario);

        // double → String
        String salarioStr = String.valueOf(salario);

        // int → double
        double valor = idade; // conversão implícita (widening)

        // double → int
        int valorInt = (int) valor; // conversão explícita (casting)

        System.out.println("String → int: " + idade);
        System.out.println("int → String: " + idadeStr);
        System.out.println("String → double: " + salario);
        System.out.println("double → String: " + salarioStr);
        System.out.println("int → double: " + valor);
        System.out.println("double → int: " + valorInt);
    }
